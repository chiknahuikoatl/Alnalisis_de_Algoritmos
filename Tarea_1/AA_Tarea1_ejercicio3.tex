            \SetKwInOut{Input}{input}\SetKwInOut{Output}{output}
            \Input{Un arreglo A de enteros, con $|A| \geq 2$}
            \Output{Una tupla $(m_1, m_2, M_1, M_2)$ donde $m_1$ contiene el mínimo de
                    A, $m_2$ el segundo menor elemento de A, $M_1$ contiene el
                    máximo de A y $M_2$ contiene el segundo mayor elemento de A.
                    }
            \BlankLine
            $m_1 \leftarrow A[0]$\;
            $m_2 \leftarrow A[0]$\;
            $M_1 \leftarrow A[0]$\;
            $M_2 \leftarrow A[0]$\;
            $i \leftarrow 1$\;
            \While{$i <$ length(A)}
            {
                $n \leftarrow A[i]$\;
                \If{$n < m_1$}
                {
                    $m_2 \leftarrow m_1$\;
                    $m_1 \leftarrow n$\;
                }
                \If{$n > M_1$}
                {
                    $M_2 \leftarrow M_1$\;
                    $M_1 \leftarrow n$\;
                }
                $i \leftarrow i + 1$\;
            }
            \KwRet{$(m_1, m_2, M_1, M_2)$}\;
            \caption{minMax}
        \end{algorithm}
